```sql
-- 1. Add target_date column (two-step — safe for any environment)
-- WARNING: Safe to run only on databases with no existing confirmed contracts.
-- If alert rows already exist, set correct target_date values before adding NOT NULL.
ALTER TABLE public.alerts ADD COLUMN IF NOT EXISTS target_date date;
UPDATE public.alerts SET target_date = '1970-01-01' WHERE target_date IS NULL;
ALTER TABLE public.alerts ALTER COLUMN target_date SET NOT NULL;

-- 2. Add notice_deadline to alert_type check constraint
ALTER TABLE public.alerts DROP CONSTRAINT IF EXISTS alerts_alert_type_check;
ALTER TABLE public.alerts ADD CONSTRAINT alerts_alert_type_check
  CHECK (alert_type IN ('day_60', 'day_30', 'day_7', 'notice_deadline'));

-- 3. Unique constraint for ON CONFLICT deduplication guard
ALTER TABLE public.alerts DROP CONSTRAINT IF EXISTS alerts_contract_alert_target_unique;
ALTER TABLE public.alerts
  ADD CONSTRAINT alerts_contract_alert_target_unique
  UNIQUE (contract_id, alert_type, target_date);

-- 4. Fix was_edited trigger:
--    TG_OP='UPDATE' guard prevents initial INSERT from /api/extract setting was_edited.
--    IS DISTINCT FROM handles null confirmed_value (Not applicable resolutions).
CREATE OR REPLACE FUNCTION set_was_edited()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'UPDATE' AND (NEW.confirmed_value IS DISTINCT FROM NEW.extracted_value) THEN
    NEW.was_edited = true;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
```

```typescript
// app/api/upload/route.ts
import { NextResponse } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { createClient as createAdminClient } from "@supabase/supabase-js";
import { randomUUID } from "crypto";

export const dynamic = "force-dynamic";
const MAX_FILE_SIZE = 20 * 1024 * 1024; // 20MB

export async function POST(request: Request) {
  const sessionClient = await createClient();
  const { data: { user } } = await sessionClient.auth.getUser();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  let formData: FormData;
  try {
    formData = await request.formData();
  } catch {
    return NextResponse.json({ error: "Invalid form data" }, { status: 400 });
  }

  const file = formData.get("file") as File | null;
  if (!file) return NextResponse.json({ error: "No file provided" }, { status: 400 });

  if (file.type !== "application/pdf") {
    return NextResponse.json({ error: "Only PDF files are accepted" }, { status: 400 });
  }

  if (file.size > MAX_FILE_SIZE) {
    return NextResponse.json({ error: "File exceeds 20MB limit" }, { status: 413 });
  }

  const contractId = randomUUID();
  const filePath = `${user.id}/${contractId}/original.pdf`;
  const fileName = file.name;

  // Admin client for storage (service role — simpler in API route context)
  const adminClient = createAdminClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!
  );

  const { error: uploadError } = await adminClient.storage
    .from("contracts")
    .upload(filePath, await file.arrayBuffer(), { contentType: "application/pdf" });

  if (uploadError) {
    console.error("Storage upload error:", uploadError);
    return NextResponse.json({ error: "File upload failed" }, { status: 500 });
  }

  // name = filename minus extension (NOT NULL in schema; overwritten on confirm)
  // category = 'other' (NOT NULL in schema; overwritten on confirm)
  const { error: dbError } = await sessionClient.from("contracts").insert({
    id: contractId,
    user_id: user.id,
    name: fileName.replace(/\.pdf$/i, ""),
    category: "other",
    status: "processing",
    extraction_status: "pending",
    file_path: filePath,
    file_name: fileName,
  });

  if (dbError) {
    console.error("DB insert error:", dbError);
    return NextResponse.json({ error: "Failed to create contract record" }, { status: 500 });
  }

  return NextResponse.json({ contract_id: contractId });
}
```

```typescript
// lib/alerts.ts
export type AlertRow = {
  contract_id: string;
  user_id: string;
  alert_type: "day_60" | "day_30" | "day_7" | "notice_deadline";
  scheduled_for: string;
  target_date: string;
  status: "pending";
};

type ContractDateData = {
  id: string;
  user_id: string;
  expiry_date: string | null;
  renewal_date: string | null;
  effective_date: string | null;
  notice_period_days: number | null;
};

function addDays(dateStr: string, days: number): Date {
  const d = new Date(dateStr + "T00:00:00Z");
  d.setUTCDate(d.getUTCDate() + days);
  return d;
}

function toISODate(d: Date): string {
  return d.toISOString().slice(0, 10);
}

function tierAlerts(
  contractId: string,
  userId: string,
  targetDateStr: string,
  today: Date
): AlertRow[] {
  const tiers: Array<{ type: AlertRow["alert_type"]; offset: number }> = [
    { type: "day_60", offset: -60 },
    { type: "day_30", offset: -30 },
    { type: "day_7", offset: -7 },
  ];
  return tiers
    .map(({ type, offset }) => ({
      contract_id: contractId,
      user_id: userId,
      alert_type: type,
      scheduled_for: toISODate(addDays(targetDateStr, offset)),
      target_date: targetDateStr,
      status: "pending" as const,
    }))
    .filter((a) => new Date(a.scheduled_for + "T00:00:00Z") > today);
}

export function buildAlerts(contract: ContractDateData, today?: Date): AlertRow[] {
  const todayStart = new Date(today ?? new Date());
  todayStart.setUTCHours(0, 0, 0, 0);

  const { id, user_id, expiry_date, renewal_date, effective_date, notice_period_days } = contract;
  const alerts: AlertRow[] = [];

  if (expiry_date) {
    alerts.push(...tierAlerts(id, user_id, expiry_date, todayStart));
  }

  // Renewal alerts: when renewal_date is present and distinct from expiry_date (or expiry is null)
  if (renewal_date && renewal_date !== expiry_date) {
    alerts.push(...tierAlerts(id, user_id, renewal_date, todayStart));
  }

  if (notice_period_days && expiry_date) {
    const deadlineDate = addDays(expiry_date, -notice_period_days);
    const scheduledDate = addDays(toISODate(deadlineDate), -7);

    // Sanity check: skip only this alert if scheduled_for < effective_date
    // Use a local flag — do NOT early-return (would discard any renewal-date alerts already pushed)
    const disqualifiedByEffectiveDate =
      !!effective_date && scheduledDate <= new Date(effective_date + "T00:00:00Z");

    // Global skip rule: skip if scheduled_for <= today (applies to all alert types)
    if (!disqualifiedByEffectiveDate && scheduledDate > todayStart) {
      alerts.push({
        contract_id: id,
        user_id: user_id,
        alert_type: "notice_deadline",
        scheduled_for: toISODate(scheduledDate),
        target_date: toISODate(deadlineDate),
        status: "pending",
      });
    }
  }

  return alerts;
}
```

```typescript
// app/api/confirm/route.ts
import { NextResponse } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { buildAlerts } from "@/lib/alerts";

export const dynamic = "force-dynamic";

const VALID_FIELDS = new Set([
  "counterparty_name", "effective_date", "expiry_date", "renewal_date",
  "auto_renew", "notice_period_days", "notice_period_text", "contract_value",
]);

export async function POST(request: Request) {
  const sessionClient = await createClient();
  const { data: { user } } = await sessionClient.auth.getUser();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  const userId = user.id;

  const { contract_id, name, category, fields } = await request.json();

  const invalidKeys = Object.keys(fields ?? {}).filter((k) => !VALID_FIELDS.has(k));
  if (invalidKeys.length > 0) {
    return NextResponse.json({ error: `Invalid field names: ${invalidKeys.join(", ")}` }, { status: 400 });
  }

  const { data: contract, error: contractError } = await sessionClient
    .from("contracts")
    .select("id, status, expiry_date, renewal_date, effective_date, notice_period_days")
    .eq("id", contract_id)
    .single();

  if (contractError || !contract) return NextResponse.json({ error: "Contract not found" }, { status: 404 });
  if (contract.status === "confirmed") return NextResponse.json({ error: "Already confirmed" }, { status: 409 });

  // Upsert extractions (skip confidence metadata row — already blocked by VALID_FIELDS check)
  const extractionRows = Object.entries(fields as Record<string, unknown>).map(
    ([fieldName, value]) => ({
      contract_id,
      field_name: fieldName,
      confirmed_value: value != null ? String(value) : null,
    })
  );

  if (extractionRows.length > 0) {
    const { error: upsertError } = await sessionClient
      .from("contract_extractions")
      .upsert(extractionRows, { onConflict: "contract_id,field_name" });
    if (upsertError) {
      console.error("Upsert error:", upsertError);
      return NextResponse.json({ error: "Failed to save field values" }, { status: 500 });
    }
  }

  const f = fields as Record<string, unknown>;
  const autoRenew =
    f.auto_renew != null
      ? typeof f.auto_renew === "string" ? f.auto_renew === "true" : Boolean(f.auto_renew)
      : null;
  const noticePeriodDays =
    f.notice_period_days != null ? parseInt(String(f.notice_period_days), 10) : null;

  // Update contracts row BEFORE generating alerts (makes 409 guard effective on retries)
  const { error: updateError } = await sessionClient
    .from("contracts")
    .update({
      name, category, status: "confirmed", updated_at: new Date().toISOString(),
      expiry_date: f.expiry_date ?? null,
      renewal_date: f.renewal_date ?? null,
      effective_date: f.effective_date ?? null,
      auto_renew: autoRenew,
      notice_period_days: noticePeriodDays,
      notice_period_text: f.notice_period_text ?? null,
      counterparty_name: f.counterparty_name ?? null,
      contract_value: f.contract_value ?? null,
      // Do NOT update extraction_confidence — preserve AI confidence score
      // Do NOT update extraction_status — left as 'review' or 'manual'; status='confirmed' is the authority
    })
    .eq("id", contract_id);

  if (updateError) {
    console.error("Update error:", updateError);
    return NextResponse.json({ error: "Failed to update contract" }, { status: 500 });
  }

  // Pre-generate alerts using confirmed values (not DB values which may be stale)
  const alertRows = buildAlerts({
    id: contract_id,
    user_id: userId,
    expiry_date: (f.expiry_date as string | null) ?? null,
    renewal_date: (f.renewal_date as string | null) ?? null,
    effective_date: (f.effective_date as string | null) ?? null,
    notice_period_days: noticePeriodDays,
  });

  let alertCount = 0;
  if (alertRows.length > 0) {
    const { error: alertError } = await sessionClient
      .from("alerts")
      .upsert(alertRows, { onConflict: "contract_id,alert_type,target_date", ignoreDuplicates: true });
    if (!alertError) alertCount = alertRows.length;
  }

  await sessionClient.from("activity_log").insert({
    user_id: userId,
    contract_id,
    event_type: "contract_confirmed",
    metadata: { contract_id, alert_count: alertCount },
  });

  return NextResponse.json({ ok: true });
}
```

```tsx
  const [state, setState] = useState<State>({ status: "idle" });
  const inputRef = useRef<HTMLInputElement>(null);
  const router = useRouter();

  async function handleFile(file: File) {
    if (file.type !== "application/pdf") {
      setState({ status: "error", message: "Only PDF files are accepted." });
      return;
    }
    if (file.size > 20 * 1024 * 1024) {
      setState({ status: "error", message: "File must be under 20MB." });
      return;

    setState({ status: "uploading" });

    const fd = new FormData();
    fd.append("file", file);

      setState({ status: "error", message: error ?? "Upload failed" });
      return;

      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ contract_id }),
    });

    setState({ status: "idle" });
    router.refresh(); // re-run server component to include new card
  }

  return (
    <div
      onDrop={(e) => { e.preventDefault(); const f = e.dataTransfer.files[0]; if (f) handleFile(f); }}
      onDragOver={(e) => e.preventDefault()}
```

```tsx
// components/dashboard/contract-list.tsx
"use client";

import { useEffect, useRef, useState } from "react";
import { createClient } from "@/lib/supabase/client";
import ContractCard, { CardState } from "./contract-card";

type ContractRow = {
  id: string; name: string; file_name: string;
  status: string; extraction_status: string;
  extraction_confidence: number | null;
  expiry_date: string | null; renewal_date: string | null;
  updated_at: string; created_at: string;
  unresolved_count: number;
};

const POLL_INTERVAL_MS = 3000;
const TIMEOUT_MS = 90_000;

function computeCardState(c: ContractRow, timedOut: boolean): CardState {
  const today = new Date(); today.setHours(0, 0, 0, 0);

  // 1. Confirmed wins
  if (c.status === "confirmed") {
    // Use expiry_date for urgency; fall back to renewal_date for daysLeft when expiry is absent
    const dateForDays = c.expiry_date ?? c.renewal_date;
    if (!dateForDays) return { type: "confirmed", urgency: "green", expiryDate: null, daysLeft: null };
    const daysLeft = Math.ceil((new Date(dateForDays + "T00:00:00").getTime() - today.getTime()) / 86400000);
    const urgency = daysLeft <= 30 ? "red" : daysLeft <= 60 ? "amber" : "green";
    return { type: "confirmed", urgency, expiryDate: c.expiry_date, daysLeft };
  }
  // 2. Manual (set by extract route)
  if (c.extraction_status === "manual") {
    return { type: "manual", message: "Scanned PDF · Manual entry needed" };
  }
  // 3. Processing (with 90s client-side timeout)
  if (c.status === "processing") {
    if (timedOut) return { type: "manual", message: "Extraction timed out. Enter dates manually." };
    return { type: "processing" };
  }
  // 4. Ready to review
  if (c.status === "review" && c.extraction_status === "review") {
    return { type: "review", unresolvedCount: c.unresolved_count };
  }
  return { type: "processing" };
}

function sortContracts(contracts: ContractRow[]): ContractRow[] {
  const today = new Date(); today.setHours(0, 0, 0, 0);
  function group(c: ContractRow) {
    if (c.status === "confirmed") {
      if (!c.expiry_date) return 3;
      const d = Math.ceil((new Date(c.expiry_date + "T00:00:00").getTime() - today.getTime()) / 86400000);
      return d <= 30 ? 0 : d <= 60 ? 1 : 2;
    }
    if (c.status === "review" && c.extraction_status === "review") return 4;
    if (c.status === "processing") return 5;
    return 6; // manual
  }
  return [...contracts].sort((a, b) => {
    const gd = group(a) - group(b);
    if (gd !== 0) return gd;
    if (a.status === "confirmed") {
      if (!a.expiry_date && !b.expiry_date) return 0;
      if (!a.expiry_date) return 1;
      if (!b.expiry_date) return -1;
      return a.expiry_date < b.expiry_date ? -1 : 1;
    }
    if (a.status === "review") return a.updated_at > b.updated_at ? -1 : 1;
    return a.created_at > b.created_at ? -1 : 1;
  });
}

function countUnresolved(extractions: any[]): number {
  return (extractions ?? []).filter(
    (e: any) => e.field_name !== "confidence" && e.confidence < 0.9 && e.confirmed_value === null && !e.was_edited
  ).length;
}

export default function ContractList({ initialContracts }: { initialContracts: ContractRow[] }) {
  const [contracts, setContracts] = useState(initialContracts);
  const contractsRef = useRef(initialContracts); // stable ref so interval doesn't recreate on every state update
  const startTimes = useRef<Map<string, number>>(new Map());
  const timedOut = useRef<Set<string>>(new Set());

  // Keep contractsRef in sync with state (without adding contracts to any effect deps)
  contractsRef.current = contracts;

  useEffect(() => {
    // Initialise start times for any processing contracts present at mount
    contractsRef.current
      .filter((c) => c.status === "processing")
      .forEach((c) => { if (!startTimes.current.has(c.id)) startTimes.current.set(c.id, Date.now()); });

    const supabase = createClient();

    // Single stable interval — reads current contracts via ref, never recreated on state change
    const timer = setInterval(async () => {
      const current = contractsRef.current;
      const now = Date.now();

      // Track start times for any newly-appeared processing contracts
      current.filter((c) => c.status === "processing").forEach((c) => {
        if (!startTimes.current.has(c.id)) startTimes.current.set(c.id, Date.now());
        if ((now - (startTimes.current.get(c.id) ?? now)) >= TIMEOUT_MS) timedOut.current.add(c.id);
      });

      const pollIds = current
        .filter((c) => c.status === "processing" && !timedOut.current.has(c.id))
        .map((c) => c.id);

      if (pollIds.length === 0) return; // nothing to poll (timed-out cards show manual state via timedOut ref)

      const { data } = await supabase
        .from("contracts")
        .select("id, status, extraction_status, extraction_confidence, expiry_date, renewal_date, updated_at, created_at, name, file_name, contract_extractions(confidence, confirmed_value, was_edited, field_name)")
        .in("id", pollIds);

      if (!data) return;

      setContracts((prev) => {
        const map = new Map(prev.map((c) => [c.id, c]));
        data.forEach((updated: any) => {
          const { contract_extractions, ...rest } = updated;
          map.set(updated.id, { ...map.get(updated.id)!, ...rest, unresolved_count: countUnresolved(contract_extractions) });
        });
        return Array.from(map.values());
      });
    }, POLL_INTERVAL_MS);

    return () => clearInterval(timer);
  }, []); // empty deps — interval is created once and reads state via contractsRef

  const sorted = sortContracts(contracts);
  if (sorted.length === 0) {
    return <p className="text-sm text-slate-400 text-center py-8">No contracts yet. Upload your first PDF above.</p>;
  }

  return (
    <div className="flex flex-col gap-3">
      {sorted.map((c) => (
        <ContractCard key={c.id} id={c.id} name={c.name || c.file_name} cardState={computeCardState(c, timedOut.current.has(c.id))} />
      ))}
    </div>
  );
}
```

```tsx
// components/review/field-row.tsx
"use client";

import { useState } from "react";

export type FieldName =
  | "counterparty_name" | "effective_date" | "expiry_date" | "renewal_date"
  | "auto_renew" | "notice_period_days" | "notice_period_text" | "contract_value";

export type Resolution = { value: string | null; isResolved: boolean };

type Props = {
  fieldName: FieldName;
  label: string;
  extractedValue: string | null;
  confidence: number;
  wasEdited: boolean;
  confirmedValue: string | null;
  resolution: Resolution;
  onResolve: (value: string | null) => void;
};

function colorState(confidence: number, wasEdited: boolean, confirmedValue: string | null) {
  if (wasEdited || confirmedValue !== null) return "blue";
  if (confidence >= 0.9) return "green";
  if (confidence >= 0.7) return "amber";
  return "red";
}

const BORDERS = { blue: "border-blue-400", green: "border-green-300", amber: "border-amber-300", red: "border-red-300" };
const BGS    = { blue: "bg-blue-50",    green: "bg-green-50",    amber: "bg-amber-50",    red: "bg-red-50" };
const LABELS = { blue: "text-blue-700", green: "text-green-700", amber: "text-amber-700", red: "text-red-700" };

export default function FieldRow({ fieldName, label, extractedValue, confidence, wasEdited, confirmedValue, resolution, onResolve }: Props) {
  const color = resolution.isResolved ? "blue" : colorState(confidence, wasEdited, confirmedValue);
  const preExpanded = (color === "amber" || color === "red") && !resolution.isResolved;
  const [editing, setEditing] = useState(preExpanded);
  const [draft, setDraft] = useState<string>(resolution.value ?? "");

  const displayVal = resolution.isResolved ? resolution.value : (color === "blue" ? confirmedValue : extractedValue);

  return (
    <div className={`border ${BORDERS[color]} rounded-lg p-3 ${BGS[color]}`}>
      <div className={`text-xs font-bold uppercase tracking-wide ${LABELS[color]} mb-1.5`}>{label}</div>

      {!editing ? (
        <div className="flex items-center justify-between gap-2">
          <span className="text-sm text-slate-700">
            {displayVal ?? <span className="text-slate-400 italic">Not found in contract</span>}
          </span>
          <button onClick={() => { setDraft(displayVal ?? ""); setEditing(true); }} className="text-xs text-slate-400 hover:text-slate-600">✎</button>
        </div>
      ) : (
        <div className="flex flex-col gap-2 mt-1">
          {fieldName === "auto_renew" ? (
            <select value={draft} onChange={(e) => setDraft(e.target.value)} className="text-sm border border-slate-300 rounded px-2 py-1.5 bg-white">
              <option value="">— select —</option>
              <option value="true">Yes</option>
              <option value="false">No</option>
            </select>
          ) : (
            <input type="text" value={draft} onChange={(e) => setDraft(e.target.value)} placeholder="Enter value…" className="text-sm border border-slate-300 rounded px-2 py-1.5 bg-white focus:outline-none focus:ring-1 focus:ring-slate-400" />
          )}
          <div className="flex gap-2 flex-wrap">
            <button onClick={() => { onResolve(draft.trim() || null); setEditing(false); }} className="text-xs bg-slate-800 text-white rounded px-3 py-1.5 font-medium hover:bg-slate-700">Save</button>
            {/* "Looks good" only shown when extractedValue IS NOT NULL — prevents silent unresolvable state */}
            {extractedValue !== null && !resolution.isResolved && (
              <button onClick={() => { onResolve(extractedValue); setEditing(false); }} className="text-xs bg-green-600 text-white rounded px-3 py-1.5 font-medium hover:bg-green-700">Looks good ✓</button>
            )}
            <button onClick={() => { onResolve(null); setEditing(false); }} className="text-xs text-slate-500 border border-slate-200 rounded px-3 py-1.5 hover:bg-slate-50">Not applicable</button>
            <button onClick={() => setEditing(false)} className="text-xs text-slate-400 hover:text-slate-600">Cancel</button>
          </div>
        </div>
      )}
    </div>
  );
}
```

```tsx
// components/review/field-panel.tsx
"use client";

import { useState } from "react";
import FieldRow, { FieldName, Resolution } from "./field-row";

export const FIELD_ORDER: Array<{ name: FieldName; label: string }> = [
  { name: "counterparty_name",   label: "Counterparty name" },
  { name: "effective_date",      label: "Effective date" },
  { name: "expiry_date",         label: "Expiry date" },
  { name: "renewal_date",        label: "Renewal date" },
  { name: "auto_renew",          label: "Auto renew" },
  { name: "notice_period_days",  label: "Notice period (days)" },
  { name: "notice_period_text",  label: "Notice period" },
  { name: "contract_value",      label: "Contract value" },
];

type ExtractionRow = {
  field_name: string;
  extracted_value: string | null;
  confirmed_value: string | null;
  confidence: number;
  was_edited: boolean;
};

type Props = {
  name: string; category: string;
  extractions: ExtractionRow[];
  onConfirm: (p: { name: string; category: string; fields: Record<string, string | null> }) => void;
  isConfirming: boolean;
};

export default function FieldPanel({ name: initName, category: initCat, extractions, onConfirm, isConfirming }: Props) {
  const [name, setName] = useState(initName);
  const [category, setCategory] = useState(initCat);

  // Initialize resolution.value from confirmed_value (if previously resolved in DB) or extracted_value.
  // For "Not applicable" (was_edited=true, confirmed_value=null), use null explicitly.
  const [resolutions, setResolutions] = useState<Record<FieldName, Resolution>>(() => {
    const map = {} as Record<FieldName, Resolution>;
    for (const { name: fn } of FIELD_ORDER) {
      const row = extractions.find((e) => e.field_name === fn);
      const wasResolvedInDb = !!(row?.was_edited || row?.confirmed_value !== null);
      map[fn] = {
        value: wasResolvedInDb ? (row!.confirmed_value) : (row?.extracted_value ?? null),
        isResolved: false,
      };
    }
    return map;
  });

  function resolve(fn: FieldName, value: string | null) {
    setResolutions((p) => ({ ...p, [fn]: { value, isResolved: true } }));
  }

  // Fields that need explicit user action: amber/red, not yet resolved in DB or this session
  const unresolvedAmberRed = FIELD_ORDER.filter(({ name: fn }) => {
    const row = extractions.find((e) => e.field_name === fn);
    if (!row) return false;
    if (row.was_edited || row.confirmed_value !== null) return false; // resolved in DB
    if (row.confidence >= 0.9) return false; // green — no action needed
    return !resolutions[fn].isResolved;
  });

  function handleConfirm() {
    const fields: Record<string, string | null> = {};
    for (const { name: fn } of FIELD_ORDER) {
      const row = extractions.find((e) => e.field_name === fn);
      const wasResolvedInDb = !!(row?.was_edited || row?.confirmed_value !== null);
      if (resolutions[fn].isResolved) {
        fields[fn] = resolutions[fn].value;           // user resolved in this session
      } else if (wasResolvedInDb) {
        fields[fn] = row!.confirmed_value;            // previously confirmed — preserve
      } else {
        fields[fn] = row?.extracted_value ?? null;    // untouched green field
      }
    }
    onConfirm({ name, category, fields });
  }

  return (
    <div className="flex flex-col gap-4">
      <div className="flex flex-col gap-1">
        <label className="text-xs font-bold uppercase tracking-wide text-slate-500">Contract name</label>
        <input type="text" value={name} onChange={(e) => setName(e.target.value)} className="text-sm border border-slate-300 rounded px-2 py-1.5 focus:outline-none focus:ring-1 focus:ring-slate-400" />
      </div>
      <div className="flex flex-col gap-1">
        <label className="text-xs font-bold uppercase tracking-wide text-slate-500">Category</label>
        <select value={category} onChange={(e) => setCategory(e.target.value)} className="text-sm border border-slate-300 rounded px-2 py-1.5 bg-white">
          <option value="saas">SaaS</option>
          <option value="lease">Lease</option>
          <option value="vendor">Vendor</option>
          <option value="employment">Employment</option>
          <option value="other">Other</option>
        </select>
      </div>

      {FIELD_ORDER.map(({ name: fn, label }) => {
        const row = extractions.find((e) => e.field_name === fn);
        return (
          <FieldRow key={fn} fieldName={fn} label={label}
            extractedValue={row?.extracted_value ?? null}
            confidence={row?.confidence ?? 0}
            wasEdited={row?.was_edited ?? false}
            confirmedValue={row?.confirmed_value ?? null}
            resolution={resolutions[fn]}
            onResolve={(v) => resolve(fn, v)}
          />
        );
      })}

      <div className="sticky bottom-0 pt-3 bg-white/90 backdrop-blur-sm">
        <button
          onClick={handleConfirm}
          disabled={unresolvedAmberRed.length > 0 || isConfirming}
          title={unresolvedAmberRed.length > 0 ? "Resolve highlighted fields first" : undefined}
          className="w-full bg-slate-900 text-white rounded-lg py-3 text-sm font-semibold hover:bg-slate-800 disabled:opacity-40 disabled:cursor-not-allowed transition-colors"
        >
          {isConfirming ? "Confirming…" : "Confirm & activate alerts"}
        </button>
        {unresolvedAmberRed.length > 0 && (
          <p className="text-xs text-center text-slate-400 mt-1.5">Resolve highlighted fields first</p>
        )}
      </div>
    </div>
  );
}
```

```tsx
// components/review/review-client.tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import Link from "next/link";
import FieldPanel from "./field-panel";

type Contract = {
  id: string; name: string; file_name: string; category: string;
  status: string; extraction_confidence: number | null;
};
type ExtractionRow = {
  field_name: string; extracted_value: string | null; confirmed_value: string | null;
  confidence: number; was_edited: boolean;
};

export default function ReviewClient({ contract, extractions, pdfUrl }: {
  contract: Contract; extractions: ExtractionRow[]; pdfUrl: string | null;
}) {
  const router = useRouter();
  const [confirming, setConfirming] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const isReadOnly = contract.status === "confirmed";

  // Widen PDF column when overall confidence < 0.70 (spec: 60/40 split)
  const lowConf = (contract.extraction_confidence ?? 1) < 0.7;
  const pdfClass    = lowConf ? "md:w-3/5" : "md:w-1/2";
  const fieldsClass = lowConf ? "md:w-2/5" : "md:w-1/2";

  async function handleConfirm(payload: { name: string; category: string; fields: Record<string, string | null> }) {
    setConfirming(true); setError(null);
    try {
      const res = await fetch("/api/confirm", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ contract_id: contract.id, ...payload }),
      });
      if (!res.ok) {
        setError((await res.json().catch(() => ({}))).error ?? "Confirm failed");
        return;
      }
      router.push("/dashboard");
    } finally {
      setConfirming(false);
    }
  }

  return (
    <div className="flex flex-col md:flex-row h-screen overflow-hidden">
      {/* PDF panel — desktop only */}
      <div className={`hidden md:block ${pdfClass} h-full border-r border-slate-200 flex-shrink-0`}>
        {pdfUrl
          ? <iframe src={pdfUrl} className="w-full h-full" title="Contract PDF" />
          : <div className="flex items-center justify-center h-full text-sm text-slate-400">PDF unavailable</div>
        }
      </div>

      {/* Fields panel */}
      <div className={`${fieldsClass} h-full overflow-y-auto`}>
        <div className="p-5">
          <Link href="/dashboard" className="text-xs text-slate-400 hover:text-slate-600 mb-4 inline-block">← Back to contracts</Link>
          <h1 className="text-lg font-semibold text-slate-900 mb-5">{contract.name || contract.file_name}</h1>

          {isReadOnly && (
            <div className="rounded-xl bg-green-50 border border-green-200 p-4 mb-5 text-sm text-green-800">
              <strong>This contract has been confirmed.</strong> Alerts are active.{" "}
              <Link href="/dashboard" className="underline">Back to contracts</Link>
            </div>
          )}

          {error && (
            <div className="rounded-lg bg-red-50 border border-red-200 p-3 mb-4 text-sm text-red-700">{error}</div>
          )}

          {isReadOnly ? (
            <div className="flex flex-col gap-3">
              {extractions.map((e) => (
                <div key={e.field_name} className="border border-slate-200 rounded-lg p-3 bg-slate-50">
                  <div className="text-xs font-bold uppercase tracking-wide text-slate-400 mb-1">{e.field_name.replace(/_/g, " ")}</div>
                  <div className="text-sm text-slate-700">{e.confirmed_value ?? <span className="italic text-slate-400">Not set</span>}</div>
                </div>
              ))}
            </div>
          ) : (
            <FieldPanel
              name={contract.name || contract.file_name}
              category={contract.category}
              extractions={extractions}
              onConfirm={handleConfirm}
              isConfirming={confirming}
            />
          )}
        </div>
      </div>
    </div>
  );
}
```
